    <script>
        (function() {
            // =====================================================
            // SIMULACIÓN DE AMIGOS Y NIVELES
            // =====================================================
            const friendsData = [
                { name: "Carlos", initialLevel: 12, avatar: "🎮" },
                { name: "Ana", initialLevel: 9, avatar: "⚡" },
                { name: "Luis", initialLevel: 6, avatar: "🔥" },
                { name: "María", initialLevel: 15, avatar: "👑" },
                { name: "Javi", initialLevel: 7, avatar: "🚀" }
            ];

            // Nivel actual del jugador (empieza en 5, como en diseño)
            let playerCurrentLevel = 5;

            // Referencias a elementos del DOM
            const playerLevelSpan = document.getElementById('playerLevel');
            const clickButton = document.getElementById('clickButton');
            const clickHint = document.getElementById('clickHint');
            const friendListDiv = document.getElementById('friendList');

            // Función para renderizar la tabla de amigos
            function renderFriends() {
                let htmlStr = '';
                friendsData.forEach(friend => {
                    htmlStr += `
                        <div class="friend-item">
                            <div class="friend-left">
                                <div class="friend-avatar">${friend.avatar}</div>
                                <div class="friend-info">
                                    ${friend.name} <small>· nivel</small>
                                </div>
                            </div>
                            <div class="friend-score">${friend.initialLevel}</div>
                        </div>
                    `;
                });
                // Añadir al jugador mismo al final, con etiqueta "tú"
                htmlStr += `
                    <div class="friend-item" style="background:#1e3155; border: 1px solid #ffdb8e;">
                        <div class="friend-left">
                            <div class="friend-avatar" style="background:#b57c1e;">👤</div>
                            <div class="friend-info" style="color:#ffdb8e;">
                                tú <span class="you-tag">TÚ</span>
                            </div>
                        </div>
                        <div class="friend-score" style="color:#ffffff; background:#2f3f68;" id="yourScoreDisplay">${playerCurrentLevel}</div>
                    </div>
                `;
                friendListDiv.innerHTML = htmlStr;
            }

            // Actualiza solo tu nivel en la tabla (para no re-renderizar todo)
            function updateYourLevelDisplay() {
                const yourScoreSpan = document.getElementById('yourScoreDisplay');
                if (yourScoreSpan) {
                    yourScoreSpan.innerText = playerCurrentLevel;
                }
                playerLevelSpan.innerText = playerCurrentLevel;
            }

            // Mecánica: subir de nivel al hacer click
            function handleClick() {
                // Incrementa nivel
                playerCurrentLevel++;
                // Actualiza displays
                playerLevelSpan.innerText = playerCurrentLevel;
                updateYourLevelDisplay();

                // Feedback rápido (opcional)
                clickHint.innerText = '¡Has escalado! +1 nivel 💪';
                setTimeout(() => {
                    clickHint.innerText = '⚡ Pulsa para escalar ⚡';
                }, 300);

                // (Opcional: podrías simular que un amigo sube de vez en cuando,
                //  pero lo dejamos simple y adictivo.)
            }

            // Función para simular invitación (viral)
            function inviteFriends() {
                alert('📲 ¡Comparte este enlace con tus amigos!\n\nhttps://tusitio.github.io/Ascend/ \n\n¡Que vean quién es el mejor!');
                // Aquí podrías abrir un enlace real de compartir, pero con alert es más simple y claro.
                // En el futuro, se podría implementar Web Share API.
            }

            // =====================================================
            // INICIALIZAR JUEGO
            // =====================================================
            function initGame() {
                // Renderiza la lista de amigos fija (más tú)
                renderFriends();

                // Configurar evento click en el botón de escalar
                clickButton.addEventListener('click', handleClick);

                // Botón de invitar
                const inviteBtn = document.getElementById('inviteBtn');
                inviteBtn.addEventListener('click', inviteFriends);

                // Para evitar doble scroll o problemas táctiles
                // Aseguramos nivel inicial en displays
                playerLevelSpan.innerText = playerCurrentLevel;
            }

            // Arrancar cuando el DOM esté listo
            window.addEventListener('DOMContentLoaded', initGame);
        })();
    </script>
